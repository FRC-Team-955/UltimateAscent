001    package core;
002    
003    import utilities.Vars;
004    import edu.wpi.first.wpilibj.DigitalInput;
005    import edu.wpi.first.wpilibj.DigitalOutput;
006    
007    /**
008     *
009     * @author RaiderPC
010     */
011    public class Compressor {
012            
013        private DigitalInput m_digInSensor = new DigitalInput(Vars.chnDigiInSensor);
014        private DigitalOutput m_digOutCompressor = new DigitalOutput(Vars.chnDigiOutCompressor);
015            
016        public void run()
017        {
018            if(m_digInSensor.get())
019                m_digOutCompressor.set(true);
020    
021            else
022                m_digOutCompressor.set(false);
023        }
024    }
