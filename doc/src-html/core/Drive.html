001package core;
002import utilities.Vars;
003import utilities.MyJoystick;
004import edu.wpi.first.wpilibj.Talon;
005import edu.wpi.first.wpilibj.Encoder;
006
007/**
008 * This is a comment!
009 * @author Jacob Payne

012
013public class Drive {        // Ramps motor speed and set motor speed 
014    
015    private String m_sDriveStatus = "Arcade Drive";
016    private Talon m_mtLeft = new Talon(utilities.Vars.chnVicDrvLeft);
017    private Talon m_mtRight = new Talon(utilities.Vars.chnVicDrvRight);
018    private Encoder m_encMotorLeft = new Encoder(Vars.chnEncMotorLeftA, Vars.chnEncMotorLeftB);
019    private Encoder m_encMotorRight = new Encoder(Vars.chnEncMotorRightA, Vars.chnEncMotorRightB);
020    private MyJoystick joy;
021    
022    /**
023     * Initializes drive, with joy and happiness.
024     * @param joystick 
025     */
026    public Drive(MyJoystick joystick) {
027        joy = joystick;
028    }
029    
030    public void run()
031    {
032        if (Vars.fnCanDrive()) 
033            arcadeDrive();
034        
035        else
036            m_sDriveStatus = "Disabled";
037        
038        Vars.fnPutDashBoardStringBox(Vars.skDriveStatus, m_sDriveStatus);
039    }
040        
041   /**
042    *  Changes the speed such that it isn't higher than it should be.
043    * @param curMtSpd
044    * @param joySpd
045    * @return 
046    */ 
047    private double ramp(double curMtSpd, double joySpd){   
048        
049        if(curMtSpd-joySpd <= -0.1)
050            return curMtSpd + 0.1;
051
052        else if(curMtSpd-joySpd >= 0.1)
053            return curMtSpd - 0.1;
054
055        return joySpd;
056    }
057    
058    /**
059     * Arcade drive. This is not tank drive
060     */
061    private void arcadeDrive(){
062        joy.setAxisChannel(MyJoystick.AxisType.kX, 3);
063        joy.setAxisChannel(MyJoystick.AxisType.kY, 2);
064        double y = joy.getY();
065        double x = joy.getX();
066        y *= Math.abs(y); // Squared Drive
067        x *= Math.abs(x); // Squared Drive
068        setSpeed(x + y, x - y);
069        //setSpeed(ramp(m_mtRight.get(), x-y), ramp(m_mtLeft.get(), x+y) );
070        m_sDriveStatus = "Arcade Drive";
071    }
072    
073    /**
074     * Returns the encoder distance on the left side of the robot.
075     * @return 
076     */
077    public double getEncoderLeftDistance(){
078        return m_encMotorLeft.getDistance();
079    }
080    
081    /**
082     * Returns the encoder distance on the right side of the robot.
083     * @return 
084     */
085    public double getEncoderRightDistance(){
086        return m_encMotorRight.getDistance();
087    }
088    
089    public void resetEncoders()
090    {
091        m_encMotorLeft.reset();
092        m_encMotorRight.reset();
093    }
094    /**
095     * Sets the motor's speed to the desired speed.
096     * @param leftMt
097     * @param rightMt 
098     */
099    public void setSpeed(double leftMt, double rightMt){
100        // Sets left and right motor speed.
101        m_mtLeft.set(leftMt);
102        m_mtRight.set(rightMt);
103    }
104}
