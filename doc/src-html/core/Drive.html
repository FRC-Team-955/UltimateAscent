012public class Drive {        // Ramps motor speed and set motor speed

014    private String m_sDriveStatus = "Arcade Drive";
015    private Victor m_mtLeft = new Victor(utilities.Vars.chnVicDrvLeft);
016    private Victor m_mtRight = new Victor(utilities.Vars.chnVicDrvRight);
017    private MyJoystick joy;

019    /**
020     * Initializes drive, with joy and happiness.
021     * @param joystick

023    public Drive(MyJoystick joystick) {
024        joy = joystick;
025    }
026    
027    public void run() {
028        if (Vars.fnCanDrive())  
029        {
030            arcadeDrive();
031            m_sDriveStatus = "Arcade Drive";
032        }
033        
034        else
035        {
036            setSpeed(0, 0);
037            m_sDriveStatus = "Disabled";
038        }
039        
040        Vars.fnPutDashBoardStringBox(Vars.skDriveStatus, m_sDriveStatus);
041    }

052   /**
053    *  Changes the speed such that it isn't higher than it should be.
054    * @param curMtSpd
055    * @param joySpd
056    * @return 
057    */ 
058    private double ramp(double curMtSpd, double joySpd){   
059        
060        
061            if(curMtSpd-joySpd <= -0.1)
062                return curMtSpd + 0.1;
063            
064            else if(curMtSpd-joySpd >= 0.1)
065                return curMtSpd - 0.1;
066            
067            return joySpd;
068        }
069    
070    /**
071     * Arcade drive. This is not tank drive
072     */
073    private void arcadeDrive(){
074        joy.setAxisChannel(MyJoystick.AxisType.kX, 1);
075        joy.setAxisChannel(MyJoystick.AxisType.kY, 3);
076        double y = joy.getY();
077        double x = joy.getX();
078        y *= Math.abs(y); // Squared Drive
079        x *= Math.abs(x); // Squared Drive
080        setSpeed(ramp(m_mtRight.get(), x-y), ramp(m_mtLeft.get(), x+y) );
081    }
082    
083    public double getMotorLeft(){
084        // Returns motor speed.
085        return m_mtLeft.get();
086    }
087    
088    public double getMotorRight(){
089        // Sets motor speed
090        return m_mtRight.get();
091    }
092    
093    public void setSpeed(double leftMt, double rightMt){
094        // Sets left and right motor speed.
095        m_mtLeft.set(leftMt);
096        m_mtRight.set(rightMt);
097        
098    }
