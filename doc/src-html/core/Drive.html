001package core;
002import utilities.Vars;
003import utilities.MyJoystick;
004import edu.wpi.first.wpilibj.Talon;
005import edu.wpi.first.wpilibj.Encoder;
006
007/**
008 * This is a comment!
009 * @author Jacob Payne

012
013public class Drive {       
014    
015    private Talon m_mtLeft = new Talon(utilities.Vars.chnVicDrvLeft);
016    private Talon m_mtRight = new Talon(utilities.Vars.chnVicDrvRight);
017    private Encoder m_encMotorLeft = new Encoder(Vars.chnEncMotorLeftA, Vars.chnEncMotorLeftB);
018    private Encoder m_encMotorRight = new Encoder(Vars.chnEncMotorRightA, Vars.chnEncMotorRightB);
019    private MyJoystick joy;
020    
021    /**
022     * Initializes drive, with joy and happiness.
023     * @param joystick 
024     */
025    public Drive(MyJoystick joystick) {
026        joy = joystick;
027    }
028    
029    public void run()
030    {
031        if (Vars.fnCanDrive()) 
032            arcadeDrive();
033    }
034        
035   /**
036    *  Changes the speed such that it isn't higher than it should be.
037    * @param curMtSpd
038    * @param joySpd
039    * @return 
040    */ 
041    private double ramp(double curMtSpd, double joySpd){   
042        
043        if(curMtSpd-joySpd <= -0.1)
044            return curMtSpd + 0.1;
045
046        else if(curMtSpd-joySpd >= 0.1)
047            return curMtSpd - 0.1;
048
049        return joySpd;
050    }
051    
052    /**
053     * Arcade drive. This is not tank drive
054     */
055    private void arcadeDrive(){
056        joy.setAxisChannel(MyJoystick.AxisType.kX, 3);
057        joy.setAxisChannel(MyJoystick.AxisType.kY, 2);
058        double y = joy.getY();
059        double x = joy.getX();
060        y *= Math.abs(y); // Squared Drive
061        x *= Math.abs(x); // Squared Drive
062        setSpeed(x + y, x - y);
063        //setSpeed(ramp(m_mtRight.get(), x-y), ramp(m_mtLeft.get(), x+y) );
064    }
065    
066    /**
067     * Returns the encoder distance on the left side of the robot.
068     * @return 
069     */
070    public double getEncoderLeftDistance(){
071        return m_encMotorLeft.getDistance();
072    }
073    
074    /**
075     * Returns the encoder distance on the right side of the robot.
076     * @return 
077     */
078    public double getEncoderRightDistance(){
079        return m_encMotorRight.getDistance();
080    }
081    
082    public void resetEncoders()
083    {
084        m_encMotorLeft.reset();
085        m_encMotorRight.reset();
086    }
087    /**
088     * Sets the motor's speed to the desired speed.
089     * @param leftMt
090     * @param rightMt 
091     */
092    public void setSpeed(double leftMt, double rightMt)
093    {
094        // Sets left and right motor speed.
095        m_mtLeft.set(leftMt);
096        m_mtRight.set(rightMt);
097    }
098}
