001package core;
002
003import edu.wpi.first.wpilibj.Encoder;
004import edu.wpi.first.wpilibj.Timer;
005import utilities.Vars;
006import utilities.MySolenoid;
007import utilities.MyPIDVelocity;
008import utilities.MyJoystick;
009import edu.wpi.first.wpilibj.Talon;
010
011/**
012 * This class is responsible for controlling the shooter and feeder.
013 * @author Fauzi
014 */
015
016// TODO: Find Velocity controller constants.
017public class Shooter {
018    
019    // CONSTANTS
020    private final double m_dSpeedIncrease = 100;
021    
022    private boolean m_bGoodToShoot = false;
023    private int m_iFrisbeeShot = 0;
024    private double m_dShootSpeed = 0;
025    private double m_dPrevPulseTime = 0;
026    private MySolenoid m_solFeeder = new MySolenoid(Vars.chnSolFeederDown, Vars.chnSolFeederUp, false);
027    private MyPIDVelocity m_PIDShooter = new MyPIDVelocity(Vars.kShooterP, Vars.kShooterI, Vars.kShooterD);
028    private Timer m_tmFeeder = new Timer();
029    private Timer m_tmPulser = new Timer();
030    private Talon m_mtShooter = new Talon(Vars.chnVicShooter);
031    private Encoder m_encShooter = new Encoder(Vars.chnEncShooterA, Vars.chnEncShooterB);
032    private MyJoystick m_joy;
033    
034    public Shooter(MyJoystick joystick)
035    {
036        m_joy = joystick;
037        m_tmPulser.start();
038    }
039    
040    public void run()
041    {
042        // When pressed, starts velocity controller so shooter motor can turn on.
043        if(m_joy.gotPressed(Vars.btShootFrisbee))
044        {
045
046            m_joy.flipSwitch(Vars.btShootFrisbee);
047            m_tmPulser.start();
048            if(!m_joy.getSwitch(Vars.btShootFrisbee))
049            {
050                m_PIDShooter.reset(true);
051                m_tmPulser.stop();
052                m_tmPulser.reset();
053            }
054        }
055
056        // Sets the shooter to the specified speed.
057        if(m_joy.getSwitch(Vars.btShootFrisbee))
058        {
059            m_mtShooter.set(m_PIDShooter.getOutput(m_dShootSpeed, getShooterEncoder()));
060
061            if(Math.abs(m_dShootSpeed - getShooterEncoder()) <= Vars.dShootTolerance)
062                m_bGoodToShoot = true;
063
064            else
065                m_bGoodToShoot = false;
066
067            // Feeds one frisbee to the shooter if shooter is at the right speed.
068            if(m_joy.gotPressed(Vars.btFeedFrisbee) && m_bGoodToShoot)
069                m_joy.flipSwitch(Vars.btFeedFrisbee);
070
071            if(m_joy.getSwitch(Vars.btFeedFrisbee))
072            {    
073                if(!m_solFeeder.getStatus() && m_bGoodToShoot)
074                {
075                    m_tmFeeder.start();
076                    m_solFeeder.turnOn();
077                }
078
079                if(m_tmFeeder.get() >= Vars.dMinFeedTime)
080                {
081                    m_solFeeder.turnOff();
082                    m_tmFeeder.stop();
083                    m_tmFeeder.reset();
084                    m_iFrisbeeShot++;
085                    m_bGoodToShoot = false;
086                }
087
088                if(m_iFrisbeeShot >= Vars.iMaxFrisbee)
089                {
090                    m_iFrisbeeShot = 0;
091                    m_joy.setSwitch(Vars.btFeedFrisbee, false);
092                }
093            }
094        }
095
096        // If we're not shooting, shooter should be zero.
097        else if(!m_joy.getSwitch(Vars.btShootFrisbee))
098            m_mtShooter.set(0);
099
100        // Sets the shooter speed when the increase or decrease button is pressed
101        if(m_joy.gotPressed(Vars.btIncreaseSpeed))
102            m_dShootSpeed += m_dSpeedIncrease;
103
104        if(m_joy.gotPressed(Vars.btDecreaseSpeed))
105            m_dShootSpeed -= m_dSpeedIncrease;
106
107        Vars.fnPutDashBoardNumberBox(Vars.skShooterSpeed, m_dShootSpeed);
108        Vars.fnPutDashBoardButton(Vars.skCanFeed, m_bGoodToShoot);
109    }
110    
111    /**
112     * Returns the shooter status.
113     * @return 
114     */
115    public double getShooterSpeed()
116    {
117        return m_dShootSpeed;
118    }
119    
120    /**
121     * Returns the shooters encoder rate.
122     * @return 
123     */
124    public double getShooterEncoder()
125    {
126        double dReturn = 0;
127        double dCurrent = m_tmPulser.get();
128        int iCount = m_encShooter.get();
129
130        m_encShooter.reset();
131        // NOTE:  60 seconds per minute;   250 counts per rotation
132        dReturn = (60.0 / 250.0) * iCount / (dCurrent - m_dPrevPulseTime);
133        m_dPrevPulseTime = dCurrent;
134        return dReturn;
135    }
136    
137    /**
138     * Returns the feeder status.
139     * @return 
140     */
141    public boolean getFeedStatus()
142    {
143        return m_joy.getSwitch(Vars.btFeedFrisbee);
144    }
145    
146    /**
147     * Sets the shooter to the desired speed.
148     * @param dSpeed 
149     */
150    public void setShooter(double dSpeed)
151    {
152        m_mtShooter.set(dSpeed);
153    }
154    
155    /**
156     * Sets the feeder based on the argument, true means feed one frisbee.
157     * @param bStatus 
158     */
159    public void setFeeder(boolean bStatus)
160    {
161        m_solFeeder.set(bStatus);
162    }
163}
