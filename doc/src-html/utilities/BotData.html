001    package utilities;
002    
003    /**
004     * Holds data corresponding to the values of the robot.
005     * @author Fauzi
006     */ 
007    public class BotData{
008        
009        private double m_dEncoderLeft = 0;
010        private double m_dEncoderRight = 0;
011        private double m_dShooterSpeed = 0;
012        private boolean m_bFeederStatus = false;
013        
014        /**
015         * Sets the values.
016         * @param dEncoderLeft
017         * @param dMtRight
018         * @param bFeedStatus 
019         */
020        public void setValues(double dEncoderLeft, double dEncoderRight, double dShootSpeed, boolean bFeedStatus)
021        {
022            m_dEncoderLeft = dEncoderLeft;
023            m_dEncoderRight = dEncoderRight;
024            m_dShooterSpeed = dShootSpeed;
025            m_bFeederStatus = bFeedStatus;
026        }
027        
028        /**
029         * Sets the values.
030         * @param bot 
031         */
032        public void setValues(Robot bot)
033        {
034            setValues(bot.getEncoderLeftDistance(), bot.getEncoderRightDistance(), bot.getShooterSpeed(), bot.getFeederStatus());
035        }
036        
037        /**
038         * Sets the values.
039         * @param emu 
040         */
041        public void setValues(BotData emu)
042        {
043            setValues(emu.getEncoderLeft(), emu.getEncoderRight(), emu.getShooterSpeed(), emu.getFeederStatus());
044        }
045        
046        /**
047         * Gets the motor value on the left side
048         * @return 
049         */
050        public double getEncoderLeft()
051        {
052            return m_dEncoderLeft;
053        }
054        
055        /**
056         * Gets the motor value on the right side.
057         * @return 
058         */
059        public double getEncoderRight()
060        {
061            return m_dEncoderRight;
062        }
063        
064        /**
065         * Gets the shooter speed.
066         * @return 
067         */
068        public double getShooterSpeed()
069        {
070            return m_dShooterSpeed;
071        }
072        
073        /**
074         * Gets the feeder status.
075         * @return 
076         */
077        public boolean getFeederStatus()
078        {
079            return m_bFeederStatus;
080        }
081    }
