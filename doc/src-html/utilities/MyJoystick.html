001package utilities;
002
003import edu.wpi.first.wpilibj.Joystick;
004
005/**
006 * Encapsulate the Joystick class so that buttons would be better.
007 * @author fauzi
008 */
009public class MyJoystick extends Joystick{
010    
011    private boolean m_bLast[];
012    private boolean m_bSwitch[];
013    private final int m_iChnVert = 6;
014    private final int m_iChnHorz = 5;
015    private final double m_dDpadTolerance = 0.2;
016    
017    public MyJoystick(int iChan, int iAmountOfButtons)
018    {
019        super(iChan);
020        m_bLast = new boolean[iAmountOfButtons];
021        m_bSwitch = new boolean[iAmountOfButtons];
022        
023        for(int index = 0; index < iAmountOfButtons; index++)
024        {
025            m_bLast[index] = false;
026            m_bSwitch[index] = false;
027        }
028    }
029    
030    /**
031     * Determines whether the specified button was pressed.
032     * @param iChan
033     * @return 
034     */
035    public boolean gotPressed(int iChan)
036    {
037        boolean bOutput = (!m_bLast[iChan] && getRawButton(iChan));
038        m_bLast[iChan] = getRawButton(iChan);
039        return bOutput;
040    }
041    
042    /**
043     * Returns the switch value. 
044     * @param iChan
045     * @return 
046     */
047    public boolean getSwitch(int iChan)
048    {
049        return m_bSwitch[iChan];
050    }
051    
052    /**
053     * Sets the switch value.
054     * @param iChan
055     * @param bVal 
056     */
057    public void setSwitch(int iChan, boolean bVal)
058    {
059        m_bSwitch[iChan] = bVal;
060    }
061    
062    /**
063     * Flips the switch value on the specified button.
064     * @param iChan 
065     */
066    public void flipSwitch(int iChan)
067    {
068        m_bSwitch[iChan] = !m_bSwitch[iChan];
069    }
070    
071    /**
072     * Gets the value of the dPad up
073     * @return 
074     */
075    public boolean getDpadUp()
076    {
077        return (getRawAxis(m_iChnVert) > m_dDpadTolerance);
078    }
079     
080    /**
081     * Gets the value of the dPad right
082     * @return 
083     */
084    public boolean getDpadRight()
085    {
086        return (getRawAxis(m_iChnHorz) > m_dDpadTolerance);
087    }
088    
089    /**
090     * Gets the value of the dPad down
091     * @return 
092     */
093    public boolean getDpadDown()
094    {
095        return (getRawAxis(m_iChnVert) < -m_dDpadTolerance);
096    }
097        
098    /**
099     * Gets the value of the dPad left
100     * @return 
101     */
102    public boolean getDpadLeft()
103    {
104        return (getRawAxis(m_iChnHorz) < -m_dDpadTolerance);
105    }
106}
