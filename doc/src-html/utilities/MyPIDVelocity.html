001    package utilities;
002    import edu.wpi.first.wpilibj.Timer;
003    
004    /**
005     *
006     * @author fauzi
007     */
008    public class MyPIDVelocity {
009       
010        // CONSTANTS
011        private final double m_kP;
012        private final double m_kI;
013        private final double m_kD;
014        
015        private double m_P = 0;
016        private double m_I = 0;
017        private double m_D = 0;
018        private double m_lastP = 0;
019        private double m_dCurrentVal = 0;
020        private double m_dLastVal = 0;
021        private double m_dOutput = 0;
022        private boolean m_bIsTimerOn = false;
023        private Timer m_tmTimer = new Timer();
024        
025        public MyPIDVelocity(double dP, double dI, double dD)
026        {
027            m_kP = dP;
028            m_kI = dI;
029            m_kD = dD;
030        }
031        
032        /**
033         * Gets output value based on arguments.
034         * @param dTarget
035         * @param dCurrent
036         * @return 
037         */
038        public double getOutput(double dTarget, double dCurrent)
039        {        
040            startTimer();
041            m_dCurrentVal = m_tmTimer.get() - m_dLastVal;
042            m_P = dTarget - dCurrent;
043            m_I = m_I + m_P * m_dCurrentVal;
044            m_D = m_P - m_lastP;
045            m_lastP = m_P;
046            m_dOutput += (m_kP * m_P) + (m_kI * m_I) + (m_kD * m_D);
047            m_dLastVal = m_tmTimer.get();
048            
049            return m_dOutput;
050        }
051        
052        /**
053         * Resets all the values, if argument is true then stop and reset time as
054         * well, other wise just leave it alone.
055         */
056        public void reset(boolean bResetTimer)
057        {
058            m_P = 0;
059            m_I = 0;
060            m_D = 0;
061            m_lastP = 0;
062            m_dCurrentVal = 0;
063            m_dLastVal = 0;
064            m_dOutput = 0;
065            
066            if(bResetTimer)
067                resetTimer();
068        }
069        
070        /**
071         * Starts the timer only if it's not active.
072         */
073        private void startTimer()
074        {
075            if(!m_bIsTimerOn)
076            {
077                m_tmTimer.start();
078                m_bIsTimerOn = true;
079            }
080        }
081        
082        /**
083         * Stops and resets the timer only if it's active.
084         */
085        private void resetTimer()
086        {
087            if(m_bIsTimerOn)
088            {
089                m_tmTimer.stop();
090                m_tmTimer.reset();
091                m_bIsTimerOn = false;
092            }
093        }
094    }
