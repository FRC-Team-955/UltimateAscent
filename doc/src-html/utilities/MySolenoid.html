001package utilities;
002
003import edu.wpi.first.wpilibj.Solenoid;
004
005/**
006 *
007 * @author Fauzi
008 */
009
010/**
011 * Purpose of this class is so that it can simply the job of making solenoids
012 * and having to manually set them individually on and off. So instead I just made
013 * a MySolenoid class with functions that do it for us :D
014 */
015public class MySolenoid 
016{
017
018    private Solenoid m_solUp;
019    private Solenoid m_solDown;
020    private boolean m_bDefaultStat;
021    
022    public MySolenoid(int solDownChannel, int solUpChannel, boolean bDefault)
023    {
024        // True means its backwards, so the default is on
025        m_bDefaultStat = bDefault;
026        m_solUp = new Solenoid(solUpChannel);
027        m_solDown = new Solenoid(solDownChannel);
028        m_solUp.set(m_bDefaultStat);
029        m_solDown.set(!m_bDefaultStat);
030    }
031    
032    /**
033     * Activates the solenoids
034     */
035    public void turnOn()
036    {
037        m_solUp.set(!m_bDefaultStat);
038        m_solDown.set(m_bDefaultStat);
039    }
040    
041    /**
042     * Deactivates the solenoids
043     */
044    public void turnOff()
045    {
046        m_solUp.set(m_bDefaultStat);
047        m_solDown.set(!m_bDefaultStat); 
048    }
049    
050    /**
051     * Sets the solenoids based on argument, true means active
052     * @param bVal 
053     */
054    public void set(boolean bVal)
055    {
056        if(bVal)
057            turnOn();
058        
059        else
060            turnOff();
061    }
062    
063    /**
064     * Returns the status of the solenoid, true means active.
065     * @return 
066     */
067    public boolean getStatus()
068    {
069        if(m_bDefaultStat)
070            return m_solDown.get();
071        
072        return m_solUp.get();
073    }
074}
