001package utilities;
002
003import core.Drive;
004import core.Shooter;
005import core.Retrieval;
006
007/**
008 * This class encapsulates the robot systems.
009 * @author fauzi
010 */
011public class Robot {
012    
013    private static Drive m_drive;
014    private static Shooter m_shooter;
015    private static Retrieval m_retrieval;
016    
017    public Robot(MyJoystick joytick)
018    {
019        m_drive = new Drive(joytick);
020        m_shooter = new Shooter(joytick);
021        m_retrieval = new Retrieval(joytick);
022    }
023    
024    /**
025     * Runs the robot code, as in Drive, Shooter.
026     */
027    public void run()
028    {
029        m_drive.run();
030        m_shooter.run();
031        m_retrieval.run();
032    }
033    
034    /**
035     * Stops the robot and sets everything to false or zero, does not disable 
036     * the ability to use it though. 
037     */
038    public void stopRobot()
039    {
040        setDriveSpeed(0, 0);
041        setShooter(0);
042        setFeeder(false);
043        setRetrieve(false);
044    }
045    
046    /**
047     * Returns the speed of the left motor.
048     * @return 
049     */
050    public double getMotorLeft()
051    {
052        return m_drive.getMotorLeft();
053    }
054    
055    /**
056     * Returns the speed of the right motor.
057     * @return 
058     */
059    public double getMotorRight()
060    {
061        return m_drive.getMotorRight();
062    }
063    
064    /**
065     * Get the speed of the shooter.
066     * @return 
067     */
068    public double getShooterSpeed()
069    {
070        return m_shooter.getShooterSpeed();
071    }
072    
073    /**
074     * Gets the feeder status, true means active.
075     * @return 
076     */
077    public boolean getFeederStatus()
078    {
079        return m_shooter.getFeedStatus();
080    }
081    
082    /**
083     * Gets the retrieve status, true means on.
084     * @return 
085     */
086    public boolean getRetrieveStatus()
087    {
088        return m_retrieval.getRetrieveStatus();
089    }
090    
091    /**
092     * Set the speed of the left and right motors, values allowed are -1 to 1.
093     * @param dLeft
094     * @param dRight 
095     */
096    public void setDriveSpeed(double dLeft, double dRight)
097    {
098        m_drive.setSpeed(dLeft, dRight);
099    }
100    
101    /**
102     * Sets the shooter to the specified speed.
103     * @param dSpeed 
104     */
105    public void setShooter(double dSpeed)
106    {
107        m_shooter.setShooter(dSpeed);
108    }
109    
110    /**
111     * Sets the feeder to true or false. True means active.
112     * @param bStatus 
113     */
114    public void setFeeder(boolean bStatus)
115    {
116        m_shooter.setFeeder(bStatus);
117    }
118    
119    /**
120     * Sets the retriever, true means active.
121     * @param bStatus 
122     */
123    public void setRetrieve(boolean bStatus)
124    {
125        m_retrieval.setRetrieve(bStatus);
126    }
127}
