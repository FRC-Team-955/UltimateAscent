091    /**
092     * Sets the double to be only at the hundreth's place, ex. 12.34.
093     */
094    public static double fnSetPrecision(double dDouble)
095    {
096        return (Double.valueOf(Math.floor(dDouble * 10 + 0.5) / 10)).doubleValue();
097    }
098    
099    /**
100     * Disables ability for user to drive robot.
101     */
102    public static void fnDisableDrive()
103    {
104        bDrive = false;
105    }
106    
107    /**
108     * Enables ability for the user to drive the robot.
109     */
110    public static void fnEnableDrive()
111    {
112        bDrive = true;
113    }
114   
115    /**
116     * Checks if the user has the ability to drive the robot.
117     */
118    public static boolean fnCanDrive()
119    {

121    }

123    /**

125     * @param iChan
