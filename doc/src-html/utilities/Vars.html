089    /**
090     * Sets the double to be only at the hundreth's place, ex. 12.34.
091     */
092    public static double fnSetPrecision(double dDouble)
093    {
094        return (Double.valueOf(Math.floor(dDouble * 10 + 0.5) / 10)).doubleValue();
095    }
096    
097    /**
098     * Disables ability for user to drive robot.
099     */
100    public static void fnDisableDrive()
101    {
102        bDrive = false;
103    }
104    
105    /**
106     * Enables ability for the user to drive the robot.
107     */
108    public static void fnEnableDrive()
109    {
110        bDrive = true;
111    }
112   
113    /**
114     * Checks if the user has the ability to drive the robot.
115     */
116    public static boolean fnCanDrive()
117    {

119    }

121    /**

123     * @param iChan
124     * @return 
125     */
126    public static boolean fnDriverGetDigitalIn(int iChan)
127    {
128        return DriverStation.getInstance().getDigitalIn(iChan);
129    }
130    
131     /**
132     * Prints specified message to the driver station on the corresponding line
133     * 2-6 are available.
134     */
135    public static void fnPrintToDriverstation(int iLine, String sMessage)
136    {
137        switch(iLine)
138        {
139            case 2:
140            {
141                DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser2, 1, "");
142                DriverStationLCD.getInstance().updateLCD(); 
143                DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser2, 1, sMessage);
144                break;
145            }
146                
147            case 3:
148            {
149                DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser3, 1, "");
150                DriverStationLCD.getInstance().updateLCD(); 
151                DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser3, 1, sMessage);
152                break;
153            }
154                
155            case 4:
156            {
157                DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser4, 1, "");
158                DriverStationLCD.getInstance().updateLCD(); 
159                DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser4, 1, sMessage);
160                break;
161            }
162                
163            case 5:
164            {
165                DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser5, 1, "");
166                DriverStationLCD.getInstance().updateLCD(); 
167                DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser5, 1, sMessage);
168                break;
169            }
170                
171            case 6:
172            {
173                DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser6, 1, "");
174                DriverStationLCD.getInstance().updateLCD(); 
175                DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser6, 1, sMessage);
176                break;
177            }
178        }
179        
180        DriverStationLCD.getInstance().updateLCD(); 
181    }
